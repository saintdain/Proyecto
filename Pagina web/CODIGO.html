int DatosHumedad;

int Bomba = 2;

void setup() {

pinMode(Bomba, OUTPUT);

pinMode(6,OUTPUT);

}

void loop()

{

Serial.println (DatosHumedad);

DatosHumedad = analogRead (A0);

if (DatosHumedad >= 650 && DatosHumedad <= 1024)

{

digitalWrite(Bomba,LOW); }

else {

digitalWrite(Bomba,HIGH);

}

}
